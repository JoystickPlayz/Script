local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer
local AutoshootEnabled = false

local function toggleAutoshoot()
    AutoshootEnabled = not AutoshootEnabled
end

local function autoshoot()
    while AutoshootEnabled do
        local tool = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Tool")
        if tool then
            tool:Activate()
        end
        RunService.Heartbeat:Wait()
    end
end

RunService.Heartbeat:Connect(function()
    if AutoshootEnabled then
        autoshoot()
    end
end)

LocalPlayer.CharacterAdded:Connect(function()
    AutoshootEnabled = false
    TextButton.Text = "Autoshoot OFF"
end)

-- Function to convert hue to RGB
local function hsvToRgb(h, s, v)
    local r, g, b
    local i = math.floor(h * 6)
    local f = h * 6 - i
    local p = v * (1 - s)
    local q = v * (1 - f * s)
    local t = v * (1 - (1 - f) * s)
    i = i % 6

    if i == 0 then
        r, g, b = v, t, p
    elseif i == 1 then
        r, g, b = q, v, p
    elseif i == 2 then
        r, g, b = p, v, t
    elseif i == 3 then
        r, g, b = p, q, v
    elseif i == 4 then
        r, g, b = t, p, v
    elseif i == 5 then
        r, g, b = v, p, q
    end

    return Color3.new(r, g, b)
end

local hue = 0 -- Initial hue value
local hueSpeed = 0.5 -- Speed of hue change

game:GetService("RunService").RenderStepped:Connect(function(deltaTime)
    -- Update hue value
    hue = hue + hueSpeed * deltaTime
    if hue > 1 then
        hue = hue - 1
    end

    -- Convert hue to RGB and apply to the note label
    local color = hsvToRgb(hue, 1, 1)
    noteLabel.TextColor3 = color
end)
